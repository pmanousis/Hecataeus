CREATE OR REPLACE TRIGGER STAGE_SPQUAL_STAGE_BIR_TRI BEFORE INSERT ON STAGE_SPQUAL_STAGE
FOR EACH ROW
DECLARE
	   tmpVar NUMBER;
/******************************************************************************
   COMPANY: ORACLE
   PROJECT: MIS
   NAME:	STAGE_SPQUAL_BIR_TRI
   PURPOSE: Default values for several columns of the table
   NOTES:

   REVISIONS:
   Ver        Date        Author           Description
   ---------  ----------  ---------------  ------------------------------------
   1.0        15/1/2007   GPAPOUTS         1. Created this trigger.
******************************************************************************/
BEGIN
	 IF :NEW.SPQUAL_SK IS NULL THEN
           SELECT STAGE_SPQUAL_STAGE_SEQ.NEXTVAL INTO tmpVar FROM dual;
          :NEW.SPQUAL_SK := tmpVar;
     END IF;
EXCEPTION
		 WHEN OTHERS THEN
-- Consider logging the error and then re-raise
   			RAISE;
END  STAGE_SPQUAL_STAGE_BIR_TRI;
/
SHOW ERRORS;
